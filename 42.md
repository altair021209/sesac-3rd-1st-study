n = int(input())
A = [0] * (1000001)

# 에라토스테네스의 체 초기화
for i in range(2, 1000001):
    A[i] = i

# 소수 판별
for i in range(2, int(1000000 ** 0.5) + 1):
    if A[i] == 0:
        continue
    for j in range(i + i, len(A), i):
        A[j] = 0

# 팰린드롬 판별 함수
def isPalindrome(target):
    temp = list(str(target))
    s = 0
    e = len(temp) - 1
    while s < e:
        if temp[s] != temp[e]:
            return False
        s += 1
        e -= 1
    return True

# 입력값 n 이상에서 소수이면서 팰린드롬인 수 찾기
i = n
while True:
    if A[i] != 0:
        result = A[i]
        if isPalindrome(result):
            print(result)
            break
    i += 1
